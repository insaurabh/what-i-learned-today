In short what spread operator (...) does in background

	From
	
	const array1 = [1,2];

	const arra2 = [3,4,5,6];

	let finalArray = [...array1, ...arra2, 7, 8, 9];

	console.log(finalArray);
	

	To
	
	var array1 = [1, 2];

	var arra2 = [3, 4, 5, 6];

	var finalArray = [].concat(array1, arra2, [7, 8, 9]);

	console.log(finalArray);
